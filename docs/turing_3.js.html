/**
 * @function
 * @name isPrimeNumber
 * @description Teste la primalité d'un entier naturel de façon basique.
 * @param {number} n - Entier naturel à tester
 * @returns {boolean} Retourne **_true_** si **_n_** est un nombre premier. Sinon, retourne **_false_**.
 */
const isPrimeNumber = (n) => {
  // Si le diviseur 'p' (compris entre 2 et √n) divise 'n', alors 'n' n'est pas
  // un nombre premier et la fonction retourne 'false'. La boucle s'arrète.
  for (let p = 2; p <= Math.sqrt(n); p++) {
    if (n % p === 0) {
      return false;
    }
  }
  // Si aucun diviseur n'a été trouvé, alors 'n' est un nombre premier. La
  // fonction retourne 'true'.
  return true;
};

const getPrimeNumberArray = (numberUpTo) => {
  let primeNumberArray = [];
  for (let i = 2; i <= Math.sqrt(numberUpTo); i++) {
    if (isPrimeNumber(i) === true) {
      primeNumberArray.push(i);
    }
  }
  return primeNumberArray;
};

const test = (number) => {
  const mesDiviseurs = getPrimeNumberArray(number);
  for (let element of mesDiviseurs) {
    if (number % element === 0) {
      return number / element;
    }
  }
};

const test1 = (number) => {
  let result = [];
  while (number / test(number) <= Math.sqrt(number)) {
    result.push(number / test(number));
    number = test(number);
  }
  result.push(number);
  return result;
};
